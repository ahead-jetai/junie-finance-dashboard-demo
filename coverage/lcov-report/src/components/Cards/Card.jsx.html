import React from 'react';
import styled from 'styled-components';
import { FiArrowUp, FiArrowDown } from 'react-icons/fi';
 
// Styled components
/**
 * A styled container component for cards, designed to provide consistent styling
 * and hover effects across the application. Utilizes theme properties for
 * background color, border radius, shadows, spacing, and transitions.
 *
 * Properties (provided through theme):
 * - `theme.background.card`: Defines the background color of the card.
 * - `theme.borderRadius.md`: The medium border-radius applied to the card edges.
 * - `theme.shadows.md`: The default shadow applied to the card.
 * - `theme.shadows.lg`: The shadow applied to the card when hovered.
 * - `theme.spacing.lg`: Specifies padding inside the card.
 *
 * Features:
 * - Full height container with flexbox layout for vertical content alignment.
 * - Hover effect that includes a subtle elevation and shadow change.
 * - Smooth transition animations for hover interactions.
 */
const CardContainer = styled.div`
  background-color: ${props => props.theme.background.card};
  border-radius: ${props => props.theme.borderRadius.md};
  box-shadow: ${props => props.theme.shadows.md};
  padding: ${props => props.theme.spacing.lg};
  height: 100%;
  display: flex;
  flex-direction: column;
  transition: transform 0.2s ease-in-out, box-shadow 0.2s ease-in-out;
 
  &:hover {
    transform: translateY(-4px);
    box-shadow: ${props => props.theme.shadows.lg};
  }
`;
 
/**
 * CardHeader is a styled div component that serves as a header
 * for a card-like UI element. It is designed to have content
 * distributed with space between items, centered alignment,
 * and a customizable bottom margin based on the theme's spacing.
 * This component adjusts its styling dynamically via props.
 */
const CardHeader = styled.div`
  display: flex;
  justify-content: space-between;
  align-items: center;
  margin-bottom: ${props => props.theme.spacing.md};
`;
 
/**
 * CardTitle is a styled-component representing an h3 HTML element.
 *
 * - It applies a font size of 1rem.
 * - The text color is dynamically derived from the `secondary` value of the theme's `text` object.
 * - No margin is applied to this element.
 *
 * This component is commonly used as a title within a card layout.
 */
const CardTitle = styled.h3`
  font-size: 1rem;
  color: ${props => props.theme.text.secondary};
  margin: 0;
`;
 
/**
 * A styled component representing a card icon container with customizable styles.
 *
 * Dynamically styled based on the theme and props provided.
 * Ensures consistent styling for icon display within a card.
 *
 * Attributes:
 * - `iconbg`: A custom background color for the icon. If not provided, the theme's primary color is used.
 *
 * Styles:
 * - Width and height are set to 36px.
 * - Circular appearance through border-radius derived from the theme.
 * - Flexbox-based center alignment for children elements.
 * - Background color is determined by the `iconbg` property or theme's primary color.
 * - Text color is based on the theme's primary text color.
 * - Default opacity is set to 0.9.
 */
const CardIcon = styled.div.attrs(props => ({
  iconbg: props.iconBg
}))`
  width: 36px;
  height: 36px;
  border-radius: ${props => props.theme.borderRadius.round};
  background-color: ${props => props.iconbg || props.theme.primary};
  display: flex;
  align-items: center;
  justify-content: center;
  color: ${props => props.theme.text.primary};
  opacity: 0.9;
`;
 
/**
 * CardContent is a styled div component with a flex value of 1.
 * It is used to provide flexible layout behavior for card components, allowing
 * the content to adapt to available space within its container.
 *
 * The styling is applied using a styled-components template literal, which
 * ensures that the component maintains its intended layout structure.
 */
const CardContent = styled.div`
  flex: 1;
`;
 
/**
 * Styled component representing the value of a card.
 *
 * This component is a styled `div` element intended to display
 * text with specific font size, font weight, and bottom margin,
 * which is dynamically adjusted based on the theme's small spacing.
 *
 * Properties:
 * - Utilizes `props.theme.spacing.sm` from the theme provider
 *   to set the bottom margin.

const CardValue = styled.div`
  font-size: 1.75rem;
  font-weight: 700;
  margin-bottom: ${props => props.theme.spacing.sm};
`;
 
/**
 * CardFooter is a styled component typically used as a footer section for card components.
 * It provides a flexible container whose items are aligned in the center.
 * The top margin is dynamically applied based on the theme's medium spacing value.

 * Properties:
 * - `theme.spacing.md`: Determines the vertical margin applied on top of the footer.

 * Usage of this component allows for consistent styling across card footers
 * with alignment and spacing based on the design system.

const CardFooter = styled.div`
  display: flex;
  align-items: center;
  margin-top: ${props => props.theme.spacing.md};
`;
 
/**
 * Styled component representing a change indicator.
 * This component is typically used to display a value change indicator that is visually styled
 * based on its positivity or negativity.
 *
 * Properties:
 * - `ispositive`: Determines if the indicator represents a positive change. If true, the color will match the success theme color.
 * - `isnegative`: Determines if the indicator represents a negative change. If true, the color will match the danger theme color.
 * - Default style: If neither `ispositive` nor `isnegative` are true, the color will default to a secondary text theme color.
 *
 * The component adjusts its appearance dynamically using the theme's specified colors and spacing values.
 */
const ChangeIndicator = styled.div.attrs(props => ({
  ispositive: props.isPositive,
  isnegative: props.isNegative
}))`
  display: flex;
  align-items: center;
  font-size: 0.875rem;
  font-weight: 600;
  color: ${props => props.ispositive 
    ? props.theme.success 
    : props.isnegative 
      ? props.theme.danger 
      : props.theme.text.secondary};
  margin-right: ${props => props.theme.spacing.md};
`;
 
/**
 * Styled component for a div element used to represent an icon that can be changed dynamically.

 * This variable creates a styled div using styled-components, with the following styles:
 * - `display`: Set to 'flex' to enable flexible box layout for its children.
 * - `align-items`: Aligned to the center along the cross-axis.
 * - `margin-right`: Dynamically set based on the theme spacing (`xs`) defined in the props.
 *
 * This component is used to add consistent styling for an icon element where necessary.

 * @constant {styled.div} ChangeIcon
 */
const ChangeIcon = styled.div`
  display: flex;
  align-items: center;
  margin-right: ${props => props.theme.spacing.xs};
`;

 * `tertiary` text color is accessed through the properties of the `props` object.

 * @type {StyledComponent}

const ChangeLabel = styled.div`
  font-size: 0.75rem;
  color: ${props => props.theme.text.tertiary};
`;
 
// Card component
/**
 * A functional React component that displays information such as a title, value, change percentage, and an optional icon.
 * This component supports formatting for numerical values and allows customization of the displayed content through children.
 *
 * @param {Object} props - The props object for the Card component.
 * @param {string} props.title - The title displayed at the top of the card.
 * @param {number|string} props.value - The main value to be displayed. Defaults to `0` if not provided or invalid.
 * @param {number|string} [props.change] - The percentage change value. Can be positive or negative. Defaults to `0` if not provided or invalid.
 * @param {string} [props.changeLabel="vs previous period"] - A label that provides context for the change value. Defaults to "vs previous period".
 * @param {ReactNode} [props.icon] - An optional icon to be displayed alongside the title.
 * @param {string} [props.iconBg] - A background color for the icon, if provided.
 * @param {function} [props.formatter] - A function to format the `value` before display. Defaults to an identity function.
 * @param {ReactNode} [props.children] - Optional additional content to be displayed in the card below the main value.
 * @returns {ReactElement} A styled card component with the provided information.
 */
const Card = ({
  title,
  value,
  change,
  changeLabel = "vs previous period",
  icon,
  iconBg,
  formatter = value => value,
  children
}) => {
  const isPositive = change > 0;
  const isNegative = change < 0;
 
  // Handle non-number values for value
  const numValue = typeof value === 'number' ? value : Number(value);
  const formattedValue = isNaN(numValue) ? formatter(0) : formatter(numValue);
 
  // Handle non-number values for change
  const numChange = typeof change === 'number' ? change : Number(change);
  const formattedChange = isNaN(numChange) ? '0.00' : Math.abs(numChange).toFixed(2);
 
  return (
    <CardContainer>
      <CardHeader>
        <CardTitle>{title}</CardTitle>
        {icon && <CardIcon iconBg={iconBg}>{icon}</CardIcon>}
      </CardHeader>
 
      <CardContent>
        <CardValue>{formattedValue}</CardValue>
        {children}
      </CardContent>
 
      {change !== undefined && (
        <CardFooter>
          <ChangeIndicator isPositive={isPositive} isNegative={isNegative}>
            <ChangeIcon>
              {isPositive ? <FiArrowUp /> : isNegative ? <FiArrowDown /> : null}
            </ChangeIcon>
            {formattedChange}%
          </ChangeIndicator>
          <ChangeLabel>{changeLabel}</ChangeLabel>
        </CardFooter>
      )}
    </CardContainer>
  );
};
 
export default Card;
